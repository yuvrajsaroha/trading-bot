import argparse
from bot.orders import place_market_order, place_limit_order
from bot.validators import validate_inputs
from bot.logging_config import setup_logging

setup_logging()

def main():
    parser = argparse.ArgumentParser()

    parser.add_argument("--symbol", required=True)
    parser.add_argument("--side", required=True)
    parser.add_argument("--type", required=True)
    parser.add_argument("--quantity", type=float, required=True)
    parser.add_argument("--price", type=float)

    args = parser.parse_args()

    try:
        validate_inputs(args.symbol, args.side, args.type, args.quantity, args.price)

        print("\n📌 Order Summary")
        print(f"Symbol: {args.symbol}")
        print(f"Side: {args.side}")
        print(f"Type: {args.type}")
        print(f"Quantity: {args.quantity}")

        if args.type == "MARKET":
            order = place_market_order(args.symbol, args.side, args.quantity)

        else:
            print(f"Price: {args.price}")
            order = place_limit_order(args.symbol, args.side, args.quantity, args.price)

        if order:
            print("\n✅ Order Placed Successfully!")
            print(f"Order ID: {order.get('orderId')}")
            print(f"Status: {order.get('status')}")
            print(f"Executed Qty: {order.get('executedQty')}")
        else:
            print("\n❌ Order Failed")

    except Exception as e:
        print(f"\n⚠️ Error: {e}")


if __name__ == "__main__":
    main()
